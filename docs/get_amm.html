 20def get_url():
 21    """
 22    Retrieves the URL used to download the file with the amm information. It' the one
 23    ending with '-utf8.zip' from the specified root URL (government Open Data)
 24
 25    Returns:
 26    str or int: If a matching URL is found, returns the URL as a string.
 27               If no matching URL is found, raise URLNotFoundError.
 28    """
 29
 30    root_url = "https://www.data.gouv.fr/fr/datasets/donnees-ouvertes-du-catalogue-e-phy-des-produits-phytopharmaceutiques-matieres-fertilisantes-et-supports-de-culture-adjuvants-produits-mixtes-et-melanges/"
 31    root_html = requests.get(root_url)
 32
 33    soup_amm = BeautifulSoup(root_html.text, features="lxml")
 34
 35    anchor_tags = soup_amm.find_all("a")
 36
 37    for tag in anchor_tags:
 38        href = tag.get("href")
 39        if href:
 40            if "-utf8.zip" in href:
 41                return href
 42
 43    raise URLNotFoundError("URL ending with '-utf8.zip' not found on the page.")

 46def download_and_extract_dataframe(
 47    amm_url, file_name="usages_des_produits_autorises_utf8.csv"
 48):
 49    """
 50    Downloads a zip file from the URL with AMM data and extracts its CSV contents into a Pandas DataFrame.
 51
 52    Args:
 53        amm_url (str): The URL of the zip file to download and extract.
 54        file_name (str): The name of the CSV file to extract from the zip archive.
 55
 56    Returns:
 57        pd.DataFrame: A pandas DataFrame containing the CSV content.
 58    """
 59    response = requests.get(amm_url)
 60
 61    with ZipFile(BytesIO(response.content), "r") as zip_file:
 62        file_list = zip_file.namelist()
 63
 64        if file_name not in file_list:
 65            raise FileNameNotFoundError(
 66                f"File '{file_name}' not found in the zip archive."
 67            )
 68
 69        with zip_file.open(file_name) as csv_file:
 70            # FRENCH format --> separator is ";", set encoding too
 71            df = pd.read_csv(csv_file, sep=";", encoding="utf-8")
 72            return df

 75def clean_df_amm(df_amm):
 76    """
 77    Cleans the DataFrame of agricultural product usages (df_amm). It returns rows for
 78        products allowed in Organic Farming ("Agriculture Biologique"), it excludes
 79        marginal uses (Thrips, Esca...). Note that it excludes Black Rot as well, which could
 80        be problematic (functionally). It return products that are still allowed.
 81
 82    Args:
 83        df_amm (pd.DataFrame): The DataFrame containing agricultural product usage data.
 84
 85    Returns:
 86        pd.DataFrame: A cleaned DataFrame containing rows that meet the specified criteria.
 87
 88    Technical riteria:
 89    1. Select rows where the column "mentions autorisees" contains the term
 90       "Utilisable en agriculture biologique" (case-insensitive).
 91    2. Further filter the DataFrame to rows where the column "identifiant usage"
 92       contains the term "vigne" (case-insensitive).
 93    3. Exclude rows where the column "mentions autorisees" contains any of the terms
 94       in the list of "exclude_usages" (case-insensitive).
 95    4. Select rows where the column "etat usage" contains the term "Retrait" (case-insensitive).
 96    """
 97    df_bio = df_amm[
 98        df_amm["mentions autorisees"].str.contains(
 99            "agriculture biologique|production biologique", case=False, na=False
100        )
101    ]
102
103    df_bio_vigne = df_bio[
104        df_bio["identifiant usage"].str.contains("vigne", case=False, na=False)
105    ]
106    exclude_usages = [
107        "Thrips",
108        "Black rot",
109        "Bact√©rioses",
110        "Excoriose",
111        "Erinose",
112        "Cochenilles",
113        "Aleurodes",
114        "Pourriture grise",
115        "Mouches",
116        "Stad. Hivern. Ravageurs",
117        "lack dead arm",
118        "Esca",
119        "Chenilles phytophages",
120        "Eutypiose",
121        "Acariens",
122    ]
123
124    df_bio_vigne_main = df_bio_vigne[
125        ~df_bio_vigne["mentions autorisees"].str.contains(
126            "|".join(exclude_usages), case=False, na=False
127        )
128    ]
129
130    df_bio_vigne_main_authorised = df_bio_vigne_main[
131        df_bio_vigne_main["etat usage"].str.contains("Autoris√©", na=False)
132    ]
133
134    df_bio_vigne_main_authorised = df_bio_vigne_main_authorised.reset_index()
135
136    return df_bio_vigne_main_authorised

139def add_others(df_bio_vigne_main_authorised):
140    """
141    Adds a new column 'Autres' to the DataFrame based on conditions in 'condition emploi' and 'gamme usage'.
142
143    Args:
144        df_bio_vigne_main_authorised (pd.DataFrame): The DataFrame containing data.
145
146    Returns:
147        pd.DataFrame: The DataFrame with the new 'Autres' column added.
148
149    Conditions:
150    - If 'badigeon' is present in 'condition emploi', 'Autres' will contain 'badigeon'.
151    - If 'jardin' is present in 'gamme usage', 'Autres' will contain 'Jardin autoris√©'.
152    - If both conditions are met, 'Autres' will contain 'badigeon|Jardin autoris√©', separated by a pipe (|) character.
153    """
154
155    # Function to merge two columns with a pipe separator
156    def merge_columns(row):
157        result = ""
158
159        # Check for 'badigeon' in 'condition emploi'
160        if pd.notna(row["condition emploi"]) and "badigeon" in row["condition emploi"]:
161            result += "badigeon"
162
163        # Check for 'jardin' in 'gamme usage'
164        if pd.notna(row["gamme usage"]) and "jardin" in row["gamme usage"]:
165            result += "|" if result else ""
166            result += "Jardin autoris√©"
167
168        return result
169
170    df_bio_vigne_main_authorised["Autres"] = df_bio_vigne_main_authorised.apply(
171        merge_columns, axis=1
172    )
173
174    return df_bio_vigne_main_authorised

201def process_concentration(value):
202    """
203    Extracts and processes the concentration value from the 'Substances actives' column.
204
205    Args:
206        value (str): The value from the 'Substances actives' column.
207
208    Returns:
209        float or np.nan: The processed concentration value as a float, rounded to two decimal places,
210            or np.nan if an error occurs during processing.
211
212    Operation:
213    * The function extracts numeric characters from the input value using a regular expression.
214    * It checks if the value contains a percent sign ("%") or " g/". Depending on the case,
215      it removes non-numeric characters, converts the value to a float, and rounds it accordingly.
216    * If any error occurs during processing, it returns np.nan.
217    """
218
219    def remove_non_numbers(input_string):
220        # Use a regular expression to match only digits and points
221        cleaned_string = re.sub(r"[^0-9.]", "", input_string)
222        return cleaned_string
223
224    try:
225        concentration_number = "".join([char for char in value if char.isdigit()])
226
227        if "%" in value:
228            clean_concentration_number = remove_non_numbers(concentration_number)
229            concentration = round(float(clean_concentration_number) / 10, 2)
230        elif " g/" in value:
231            clean_concentration_number = remove_non_numbers(concentration_number)
232            concentration = round(float(clean_concentration_number) / 100, 2)
233        else:
234            clean_concentration_number = remove_non_numbers(concentration_number)
235            concentration = round(float(clean_concentration_number) / 100, 2)
236
237        return concentration
238
239    except:
240        return np.nan

243def make_second_names_main(df_bio_vigne_main_authorised_with_others_compounds):
244    """
245    Expand the DataFrame by creating new rows for each product name in the 'seconds noms commerciaux' column.
246    Each new line has the second name as the main name ("nom produit").
247    WARNING: this uses iterrows(). It should be fine since the volume of data is low.
248
249    Args:
250        df_bio_vigne_main_authorised_with_others_compounds (DataFrame): The input DataFrame.
251
252    Returns:
253        DataFrame: A new DataFrame with additional rows for each product name in the 'seconds noms commerciaux' column.
254    """
255    column_names = df_bio_vigne_main_authorised_with_others_compounds.columns.tolist()
256    df_second_name = pd.DataFrame(columns=column_names)
257    for _, row in df_bio_vigne_main_authorised_with_others_compounds.iterrows():
258        second_names_raw = row["seconds noms commerciaux"]
259        if pd.notna(second_names_raw):
260            second_names_list = second_names_raw.split("|")
261            for second_name in second_names_list:
262                second_name = second_name.strip()
263
264                second_row = row.copy()
265                second_row["nom produit"] = second_name
266
267                df_second_name = pd.concat(
268                    [df_second_name, second_row.to_frame().T], ignore_index=True
269                )
270
271    df_bio_vigne_main_authorised_with_others_compounds = pd.concat(
272        [df_bio_vigne_main_authorised_with_others_compounds, df_second_name],
273        ignore_index=True,
274    )
275
276    return df_bio_vigne_main_authorised_with_others_compounds

367def create_df_insecticide(df_bio_vigne_main_authorised_with_others_compounds):
368    """
369    Creates a DataFrame containing information about insecticide products.
370
371    Args:
372        df_bio_vigne_main_authorised_with_others_compounds (pd.DataFrame): Input DataFrame containing product information.
373
374    Returns:
375        pd.DataFrame: DataFrame containing insecticide product information.
376
377    The function filters the input DataFrame based on the 'fonctions' column, selecting rows where the term
378    'insecticide' is present. It then further filters based on specified active substances: 'Spinosad', 'Bacillus',
379    and 'pyr√©thrines'. The 'Concentration' column is derived from the 'Substances actives' column, and concentration
380    values are processed using the 'process_concentration' function. For products containing 'Bacillus', the concentration
381    is set to 0 by convention.
382
383    The 'Dose' column is calculated based on the concentration and 'dose retenue'. Additionally, a 'Biocontr√¥le (1/0)'
384    column is created, with 1 indicating the presence of 'biocontr√¥le' in the 'mentions autorisees', and 0 otherwise.
385
386    The resulting DataFrame includes columns: 'nom produit', 'Active Compound', 'Autres', 'dose retenue', 'Dose',
387    'Biocontr√¥le (1/0)', and 'nombre max d'application'.
388    """
389    df_insecticide = df_bio_vigne_main_authorised_with_others_compounds[
390        df_bio_vigne_main_authorised_with_others_compounds["fonctions"].str.contains(
391            "insecticide", case=False, na=False
392        )
393    ].reset_index(drop=True)
394
395    substances_to_include = ["Spinosad", "Bacillus", "pyr√©thrines"]
396
397    df_insecticide = df_insecticide[
398        df_insecticide["Substances actives"].apply(
399            lambda x: any(sub in x for sub in substances_to_include)
400        )
401    ].reset_index(drop=True)
402
403    df_insecticide["Concentration"] = (
404        df_insecticide["Substances actives"].str.split(")").str[1]
405    )
406    df_insecticide["Concentration"] = df_insecticide["Concentration"].apply(
407        process_concentration
408    )
409
410    # By convention, when the product is Bacillus the concentration is set to 0
411    df_insecticide.loc[
412        df_insecticide["Substances actives"].str.contains(
413            "Bacillus", case=False, na=False
414        ),
415        "Concentration",
416    ] = 0
417
418    df_insecticide["Dose"] = (
419        ((df_insecticide["Concentration"] / 100) * df_insecticide["dose retenue"])
420        .astype(float)
421        .round(3)
422    )
423
424    df_insecticide["Biocontr√¥le (1/0)"] = df_insecticide["mentions autorisees"].apply(
425        lambda x: 1 if "biocontr√¥le" in x.lower() else 0
426    )
427
428    df_insecticide = df_insecticide[
429        [
430            "nom produit",
431            "Active Compound",
432            "Autres",
433            "dose retenue",
434            "Dose",
435            "Biocontr√¥le (1/0)",
436            "nombre max d'application",
